~\marginpar[left]{\textbf{(۱۵ نمره)}}
جانشینی‌های زیر را انجام دهید.
\begin{enumerate}
  \item $\forall x (\forall y R(z, x + y))[x+y/z]$
  \item $\exists x (\forall y P(x, y))[x+y/x]$
\end{enumerate}\quad
\begin{ans}
  \begin{enumerate}
    \item $\forall u (\forall v R(x + y, u + v))$
    \item $\exists x (\forall y P(x, y))$
  \end{enumerate}
\end{ans}
