
بدون استفاده از قاعدهٔ \lr{RAA} برای گزاره‌های زیر درخت استنتاج بسازید:
\begin{enumerate}[label=(\alph*)]
    \item $\neg\neg\neg\varphi\dashv\vdash\neg\varphi$
    \item $\vdash\neg\neg(\varphi\vee\neg\varphi)$
    \item $\neg\neg(\varphi\to\psi)\dashv\vdash(\neg\neg\varphi\to\neg\neg\psi)$
    \item $\neg(\varphi\vee\psi)\dashv\vdash(\neg\varphi\wedge\neg\psi)$
    \item $\varphi\vee\psi\vdash\neg(\neg\varphi\wedge\neg\psi)$
    \item $\vdash\neg\neg(\neg\neg\varphi\to\varphi)$
\end{enumerate}\quad
\begin{ans}
    \begin{enumerate}[label=(\alph*)]
        \item\nopagebreak\quad\\\LTR
        \begin{prooftree}
            \AxiomC{$[\varphi]^2$}
            \AxiomC{$[\neg\varphi]^1$}
            \RightLabel{$\bot$\lr{I}}\BinaryInfC{$\bot$}
            \RightLabel{$\neg$\lr{I} $(1)$}\UnaryInfC{$\neg\neg\varphi$}
            \AxiomC{$\neg\neg\neg\varphi$}
            \RightLabel{$\bot$\lr{I}}\BinaryInfC{$\bot$}
            \RightLabel{$\neg$\lr{I} $(2)$}\UnaryInfC{$\neg\varphi$}
        \end{prooftree}
        \begin{prooftree}
            \AxiomC{$\neg\varphi$}
            \AxiomC{$[\neg\neg\varphi]^1$}
            \RightLabel{$\bot$\lr I}\BinaryInfC{$\bot$}
            \RightLabel{$\neg$\lr I $(1)$}\UnaryInfC{$\neg\neg\neg\varphi$}
        \end{prooftree}\RTL
        \item\quad\\\LTR
        \begin{prooftree}
            \AxiomC{$[\varphi]^1$}
            \RightLabel{$\vee$\lr I}\UnaryInfC{$\varphi\vee\neg\varphi$}
            \AxiomC{$[\neg(\varphi\vee\neg\varphi)]^2$}
            \RightLabel{$\bot$\lr I}\BinaryInfC{$\bot$}
            \RightLabel{$\neg$\lr I $(1)$}\UnaryInfC{$\neg\varphi$}
            \RightLabel{$\vee$\lr I}\UnaryInfC{$\varphi\vee\neg\varphi$}

            \RightLabel{$\bot$\lr I}\BinaryInfC{$\bot$}

        \end{prooftree}\RTL
        \item\quad\\\LTR

            \AxiomC{$[\varphi\to\psi]^2$}

            \AxiomC{$[\psi]^3$}
            \RightLabel{$\bot$\lr I}\BinaryInfC{$\bot$}
            \RightLabel{$\neg$\lr I $(1)$}\UnaryInfC{$\neg\varphi$}
            \AxiomC{$[\neg\neg\varphi]^4$}
            \RightLabel{$\bot$\lr I}\BinaryInfC{$\bot$}
            \RightLabel{$\neg$\lr I $(2)$}\UnaryInfC{$\neg(\varphi\to\psi)$}

            \RightLabel{$\bot$\lr I}\BinaryInfC{$\bot$}
            \RightLabel{$\neg$\lr I $(3)$}\UnaryInfC{$\neg\neg\psi$}
            \RightLabel{$\to$\lr I $(4)$}\UnaryInfC{$\neg\neg\varphi\to\neg\neg\psi$}
        \end{prooftree}

            \RightLabel{$\bot$\lr I}\BinaryInfC{$\bot$}

            \AxiomC{$[\neg\varphi]^3$}

            \RightLabel{$\bot$\lr I}\BinaryInfC{$\bot$}
            \RightLabel{$\neg$\lr I $(2)$}\UnaryInfC{$\neg\varphi$}

            \RightLabel{$\bot$\lr E}\UnaryInfC{$\psi$}
            \RightLabel{$\to$\lr I $(4)$}\UnaryInfC{$\varphi\to\psi$}
            
            \AxiomC{$[\neg(\varphi\to\psi)]^5$}
            \RightLabel{$\bot$\lr I}\BinaryInfC{$\bot$}
            \RightLabel{$\neg$\lr I $(5)$}\UnaryInfC{$\neg\neg(\varphi\to\psi)$}

        \item\quad\\\LTR
        \begin{prooftree}

            \veeI{$\varphi\vee\psi$}
            \AxiomC{$\neg(\varphi\vee\psi)$}
            \botI

            \AxiomC{$[\psi]^2$}
            \veeI{$\varphi\vee\psi$}
            \AxiomC{$\neg(\varphi\vee\psi)$}

            \conI{$\neg\varphi\wedge\neg\psi$}

        \end{prooftree}

            \AxiomC{$[\varphi]^1$}
            \AxiomC{$\neg\varphi\wedge\neg\psi$}

        می‌سازیم. سپس با استفاده از بخش
        (؟)

        می‌سازیم. با ضمیمه کردن دو استنتاج به یکدیگر استنتاج مطلوب حاصل می‌شود.
    \end{enumerate}
